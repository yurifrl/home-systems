6#   - Win11 24H2 retail amd64 base (built via UUP dump)

8#   - autounattend.xml at the ISO root (hands-free install)

hack/win11-iso/sign-url.sh
Mints a ≤7-day V4 signed URL for the private bucket object and prints a paste-ready values snippet. SA key via mktemp+trap (never left on disk).
1#!/usr/bin/env bash
2# win11-iso-sign-url.sh — generate a V4 signed GET URL for the Win11 combined
3# ISO in the PRIVATE iso-images GCS bucket, for Proxmox to download via the
4# crossplane-proxmox `EnvironmentDownloadFile`.
5#
6# WHY THIS EXISTS
7#   The iso-images bucket is private (public-access-prevention ENFORCED), so
8#   Proxmox can't pull a plain URL. Crossplane's download resource only uses the
9#   URL for the ONE-TIME download — once the ISO is resident on Proxmox the MR
10#   stays Ready even after the URL expires (verified empirically). So you only
11#   need a fresh URL when the ISO must be (re)downloaded:
12#     - first deploy of the Windows VM, or
13#     - a Proxmox host wipe/reinstall, or
14#     - the previous signed URL is older than its TTL (max 7 days).
15#
16# WHAT IT DOES
17#   - pulls the GCS service-account key from 1Password (item: crossplane-gcp)
18#   - mints a V4 signed URL (default 7 days) for the ISO object
19#   - prints the URL and a ready-to-paste snippet for the PRIVATE values repo
20#
21# USAGE
22#   .bin/.. or: bash hack/win11-iso/sign-url.sh [days]
23#
24# AFTER RUNNING
25#   Paste the URL into the PRIVATE values repo (home-systems-values):
26#     proxmox/values.yaml -> isos.win11.url
27#   commit + push; ArgoCD/Crossplane re-pulls the ISO to Proxmox.
28#
29# REQUIREMENTS: op (1Password CLI, signed in), python3 with google-cloud-storage
30#   (pip install google-cloud-storage), network to GCS.
31set -euo pipefail
32
33DAYS="${1:-7}"
34PROJECT="syscd-443112"
35BUCKET="syscd-iso-images-4c4398cd-1a7"      # iso-images (private) — see home-systems-values/gcp/values.yaml
36OBJECT="Win11_24H2_combined.iso"
37KEYFILE="$(mktemp)"
38trap 'rm -f "$KEYFILE"' EXIT
39
40echo "# fetching GCS SA key from 1Password (crossplane-gcp)..." >&2
41op item get crossplane-gcp --vault kubernetes --fields creds --reveal \
42  | python3 -c "import sys;s=sys.stdin.read().strip();s=s[1:-1] if s.startswith('\"') else s;open('$KEYFILE','w').write(s.replace('\"\"','\"'))"
43
44URL=$(python3 - "$BUCKET" "$OBJECT" "$DAYS" "$KEYFILE" "$PROJECT" <<'PY'
45import sys, datetime
46from google.cloud import storage
47from google.oauth2 import service_account
48bkt, obj, days, keyfile, project = sys.argv[1], sys.argv[2], int(sys.argv[3]), sys.argv[4], sys.argv[5]
49creds = service_account.Credentials.from_service_account_file(keyfile)
50client = storage.Client(credentials=creds, project=project)
51blob = client.bucket(bkt).blob(obj)
52print(blob.generate_signed_url(version="v4", expiration=datetime.timedelta(days=days), method="GET"))
53PY
54)
55
56echo "# signed URL for gs://$BUCKET/$OBJECT (valid ${DAYS}d):" >&2
57echo
58echo "# --- paste into home-systems-values/proxmox/values.yaml ---"
59echo "isos:"
60echo "  win11:"
61echo "    enabled: true"
62echo "    url: \"$URL\""

.bin/gpu-switch.sh
Flips the shared GPU between the two VMs by setting per-VM started/passthrough and applying via Crossplane — stop one, settle, start the other (sequenced to avoid a busy-GPU race).
1#!/usr/bin/env bash
2# gpu-switch.sh - switch ownership of the shared GPU (0000:01:00.0) between the
3# two Proxmox VMs managed by the crossplane-proxmox chart.
4#
5#   talos   -> VM 100 started (GPU passthrough), VM 101 stopped
6#   windows -> VM 101 started + GPU passthrough variant, VM 100 stopped
7#
8# The switch changes DESIRED state in Crossplane (renders the chart and applies
9# the two EnvironmentVMs); Crossplane then stops one VM and starts the other.
10# A hand `qm stop` would be undone by reconcile, so always use this.
11#
12# Usage:
13#   .bin/gpu-switch.sh talos|tl|t | windows|win|w [--apply]
14#
15# Without --apply it prints the rendered EnvironmentVMs (dry run). With --apply
16# it pipes them to `kubectl apply` against $KUBECONTEXT.
17#
18# Env:
19#   KUBECONTEXT   kubectl context where Crossplane runs (default: current)
20#   CHART         chart path (default: k8s/charts/crossplane-proxmox)
21#   EXTERNAL_SECRET  "true" to keep the chart's ExternalSecret (1Password/ESO),
22#                    "false" (default) when the proxmox-creds Secret is supplied
23#                    out-of-band (e.g. a local kind control plane).
24set -euo pipefail
25
26REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
27CHART="${CHART:-$REPO_ROOT/k8s/charts/crossplane-proxmox}"
28EXTERNAL_SECRET="${EXTERNAL_SECRET:-false}"
29
30target="${1:-}"
31apply="${2:-}"
32case "$target" in
33  talos|tl|t)
34    sets=(
35      --set vms.talosPc01.enabled=true  --set vms.talosPc01.started=true
36      --set vms.windows.enabled=true    --set vms.windows.started=false
37    ) ;;
38  windows|win|w)
39    sets=(
40      --set vms.talosPc01.enabled=true  --set vms.talosPc01.started=false
41      --set vms.windows.enabled=true    --set vms.windows.started=true
42      --set vms.windows.passthrough=true --set vms.windows.vga=null
43    ) ;;
44  *)
45    echo "usage: $(basename "$0") talos|tl|t | windows|win|w [--apply]" >&2
46    exit 2 ;;
47esac
48
49render() {
50  helm template crossplane-proxmox "$CHART" \
51    --set credentials.externalSecret.enabled="$EXTERNAL_SECRET" \
52    "${sets[@]}" \
53    --show-only templates/talos-pc01.yaml \
54    --show-only templates/windows.yaml
55}
56
57echo "# gpu-switch -> $target  (chart: $CHART)" >&2
58if [ "$apply" = "--apply" ]; then
59  ctx_args=()
60  [ -n "${KUBECONTEXT:-}" ] && ctx_args=(--context "$KUBECONTEXT")
61  K(){ kubectl ${ctx_args[@]+"${ctx_args[@]}"} "$@"; }
62  one(){ helm template crossplane-proxmox "$CHART" --set credentials.externalSecret.enabled="$EXTERNAL_SECRET" "${sets[@]}" --show-only "templates/$1"; }
63  # Sequence the switch: the shared GPU (0000:01:00.0) must be released by the
64  # VM being stopped BEFORE the other starts, or the passthrough start fails on
65  # a busy device. Apply the stop first, let it settle, then the start.
66  case "$target" in
67    talos|tl|t)    stop_tpl=windows.yaml;    start_tpl=talos-pc01.yaml ;;
68    windows|win|w) stop_tpl=talos-pc01.yaml; start_tpl=windows.yaml ;;
69  esac
70  echo "# applying to context: ${KUBECONTEXT:-<current>}" >&2
71  echo "# 1/3 stop GPU holder ($stop_tpl)" >&2; one "$stop_tpl" | K apply -f -
72  echo "# 2/3 settle ~60s so the GPU is released (provider exposes no power-state field to poll)" >&2; sleep 60
73  echo "# 3/3 start GPU owner ($start_tpl)" >&2; one "$start_tpl" | K apply -f -
74  echo "# done. VERIFY: K get environmentvm <start-vm> -o jsonpath Synced should be True;" >&2
75  echo "#   if it errored on a busy GPU, the stop hadn't finished — re-run or wait + reapply." >&2
76else
77  render
78  echo "# dry run only. re-run with --apply to switch." >&2
79fi
